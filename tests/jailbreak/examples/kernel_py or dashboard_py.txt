# Example: Security status widget in your dashboard UI
def create_security_widget(parent):
    frame = tk.Frame(parent, relief=tk.RAISED, bd=1)
    frame.pack(fill=tk.X, padx=5, pady=5)

    # Header
    tk.Label(frame, text="🔒 Security Status", font=("Arial", 12, "bold")).pack(anchor=tk.W)

    # Status label (updated periodically)
    status_label = tk.Label(frame, text="Checking...", font=("Arial", 10))
    status_label.pack(anchor=tk.W)

    def update_status():
        data = get_security_panel_data()
        status = data.get("security", {})
        status_label.config(
            text=f"{status.get('status_icon', '⚪')} {status.get('status_label', 'Unknown')} | "
                 f"Refusals: {status.get('refusals', 0)}/{status.get('total_turns', 0)} | "
                 f"Severity: {status.get('max_severity', 'NONE')}"
        )
        # Update every 30 seconds
        parent.after(30000, update_status)

    update_status()

    # Quick run button
    def run_test():
        result = run_security_test_from_dashboard("crescendo", "trinity")
        status_label.config(text=f"🔄 Running...")
        # Re-fetch after test
        parent.after(5000, update_status)

    tk.Button(frame, text="Run Security Test", command=run_test).pack(anchor=tk.W, pady=2)

    return frame
